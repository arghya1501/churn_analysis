# Example DataFrame
data = {
    'SeniorCitizen': [0, 1, 0, 1, 0, 0, 1, 0, 1, 0],
    'Churn': ['Yes', 'No', 'Yes', 'Yes', 'No', 'No', 'No', 'Yes', 'Yes', 'No']

stacked_data = (
    df.groupby(['SeniorCitizen', 'Churn'])

    .unstack(fill_value=0)

            va='center', 
            color='white', 
            fontsize=10
        )
